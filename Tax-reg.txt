const querySchema = z.object({
  type: z.enum(['INCOME_TAX', 'SALES_TAX', 'FEDERAL_EXCISE']).optional(),
  jurisdiction: z.string().optional(),
  section: z.string().optional(),
  search: z.string().optional(),
  page: z.string().default('1'),
  limit: z.string().default('20'),
})

export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url)
    const params = querySchema.parse(Object.fromEntries(searchParams))
    
    const where: any = {}
    
    if (params.type) where.type = params.type
    if (params.jurisdiction) where.jurisdiction = params.jurisdiction
    if (params.section) {
      where.section = { contains: params.section, mode: 'insensitive' }
    }
    
    if (params.search) {
      where.OR = [
        { title: { contains: params.search, mode: 'insensitive' } },
        { content: { contains: params.search, mode: 'insensitive' } },
        { plainEnglish: { contains: params.search, mode: 'insensitive' } },
      ]
    }
    
    const regulations = await prisma.regulation.findMany({
      where,
      include: {
        amendments: { orderBy: { date: 'desc' } },
        sros: { orderBy: { date: 'desc' }, take: 3 },
      },
      skip: (parseInt(params.page) - 1) * parseInt(params.limit),
      take: parseInt(params.limit),
      orderBy: { effectiveDate: 'desc' },
    })
    
    const total = await prisma.regulation.count({ where })
    
    return NextResponse.json({
      data: regulations,
      meta: {
        total,
        page: parseInt(params.page),
        pages: Math.ceil(total / parseInt(params.limit)),
      },
    })
  } catch (error) {
    return NextResponse.json(
      { error: 'Failed to fetch regulations' },
      { status: 500 }
    )
  }

model User {
  id            String    @id @default(cuid())
  email         String    @unique
  name          String?
  ntn           String?   @unique // National Tax Number
  filerStatus   Boolean   @default(false)
  province      String?
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  notices       Notice[]
  searches      SearchHistory[]
  alerts        Alert[]
  
  @@map("users")
}

model Regulation {
  id              String   @id @default(cuid())
  type            RegulationType
  jurisdiction    String   // Federal, Punjab, Sindh, KPK, Balochistan
  lawName         String   // "Income Tax Ordinance 2001"
  section         String   // "Section 122"
  subSection      String?
  clause          String?
  title           String
  content         String   @db.Text
  plainEnglish    String?  @db.Text // Simplified explanation
  effectiveDate   DateTime
  amendedDate     DateTime?
  status          Status   @default(ACTIVE)
  
  amendments      Amendment[]
  relatedCases    CaseLaw[]
  sros            SRO[]
  tags            Tag[]
  
  createdAt       DateTime @default(now())
  updatedAt       DateTime @updatedAt
  
  @@index([type, jurisdiction])
  @@index([section])
  @@index([status])
  @@map("regulations")
}

model Amendment {
  id              String   @id @default(cuid())
  regulationId    String
  regulation      Regulation @relation(fields: [regulationId], references: [id], onDelete: Cascade)
  date            DateTime
  description     String
  sroNumber       String?
  previousText    String?  @db.Text
  newText         String?  @db.Text
  
  @@map("amendments")
}

model SRO {
  id              String   @id @default(cuid())
  number          String   // "SRO 60(I)/2026"
  title           String
  summary         String   @db.Text
  date            DateTime
  effectiveDate   DateTime
  type            SROType
  issuer          String   // FBR, PRA, SRB, KPRA, BRA
  impact          ImpactLevel
  pdfUrl          String?
  
  regulations     Regulation[]
  
  @@index([number])
  @@index([date])
  @@map("sros")
}

model Notice {
  id              String   @id @default(cuid())
  userId          String
  user            User     @relation(fields: [userId], references: [id])
  
  noticeType      NoticeType
  fbrReference    String?  // FBR tracking number
  taxYear         String
  section         String   // Section under which issued
  issueDate       DateTime
  dueDate         DateTime?
  
  // Parsed data
  amounts         Json?    // { demand: "2450000", penalty: "245000", surcharge: "367500" }
  deadlines       Json?    // [{ date: "2026-03-15", action: "File Appeal", section: "127" }]
  actions         Json?    // [{ title: "Review Order", description: "...", form: "IT-01" }]
  citations       String[] // Cited regulations
  
  rawText         String   @db.Text
  analyzed        Boolean  @default(false)
  
  createdAt       DateTime @default(now())
  @@map("notices")
}

model SearchHistory {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id])
  query       String
  filters     Json?
  results     Int
  createdAt   DateTime @default(now())
  @@map("search_history")
}

model CaseLaw {
  id            String   @id @default(cuid())
  citation      String   // "2024 PTD 1234"
  court         String   // "Supreme Court", "High Court"
  date          DateTime
  title         String
  summary       String   @db.Text
  fullText      String   @db.Text
  regulations   Regulation[]
  createdAt     DateTime @default(now())
  @@map("case_law")
}

model Tag {
  id          String   @id @default(cuid())
  name        String   @unique
  regulations Regulation[]
  @@map("tags")
}

                const matchesJurisdiction = !this.demoFilters.jurisdiction || reg.jurisdiction === this.demoFilters.jurisdiction;
                const matchesSearch = !this.demoFilters.search || 
                    reg.title.toLowerCase().includes(this.demoFilters.search.toLowerCase()) ||
